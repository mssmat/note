        function isPrime(num) {
            if (num <= 1) return false;
            if (num <= 3) return true;
            if (num % 2 === 0 || num % 3 === 0) return false;
            for (let i = 5; i * i <= num; i += 6) {
                if (num % i === 0 || num % (i + 2) === 0) return false;
            }
            return true;
        }

        let currentNumber = 2;
        function displayNextPrime() {
            while (!isPrime(currentNumber)) {
                currentNumber++;
            }
            const primeContainer = document.getElementById('prime-container');
            primeContainer.innerHTML = ''; // Clear previous prime number
            const primeElement = document.createElement('div');
            primeElement.className = 'prime-number';
            primeElement.textContent = currentNumber;
            primeContainer.appendChild(primeElement);

            // Change text color from light to dark and back
            let opacity = 0.2;
            const interval = setInterval(() => {
                opacity += 0.1;
                primeElement.style.color = `rgba(226, 232, 240, ${opacity})`;
                if (opacity >= 1) {
                    clearInterval(interval);
                    setTimeout(() => {
                        let fadeOutInterval = setInterval(() => {
                            opacity -= 0.1;
                            primeElement.style.color = `rgba(226, 232, 240, ${opacity})`;
                            if (opacity <= 0.2) {
                                clearInterval(fadeOutInterval);
                                currentNumber++;
                                setTimeout(displayNextPrime, 1000);
                            }
                        }, 100);
                    }, 1000);
                }
            }, 100);
        }

        displayNextPrime();
    </script>
</body>
</html>
python 
import time

def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

def prime_generator():
    num = 2
    while True:
        if is_prime(num):
            yield num
        num += 1

def display_primes():
    for prime in prime_generator():
        print(prime)
        time.sleep(1)

if __name__ == "__main__":
    display_primes()
