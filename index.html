console.error("関数引数のパースに失敗:", parseError);
console.error("受信した引数文字列:", toolCall.function.arguments);
showError("AIからの応答データの解析に失敗しました");
responseText.innerHTML = "";
responseContainer.style.display = "none";
}
} else {
console.warn("想定外の関数が呼び出されました:", toolCall.function?.name);
showError("AIが想定外の動作をしました");
responseText.innerHTML = "";
responseContainer.style.display = "none";
}
} else if (message.content) {
const aiResponse = message.content;
responseText.innerHTML = `${aiResponse}`;
responseContainer.style.display = "block";
errorMessage.style.display = "none";
console.log("テキスト応答受信:", aiResponse);
} else {
console.warn("Tool呼び出しもテキスト応答もありませんでした", data);
showError("AIから有効な応答が得られませんでした");
responseText.innerHTML = "";
responseContainer.style.display = "none";
}

} catch (error) {
console.error("エラー:", error);
showError(error.message || "不明なエラーが発生しました");
responseText.innerHTML = "";
responseContainer.style.display = "none";
} finally {
submitBtn.disabled = false;
loadingElement.style.display = "none";
}
});

function showError(message) {
errorMessage.textContent = message;
errorMessage.style.display = "block";
}

});
